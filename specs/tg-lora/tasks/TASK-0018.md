### 1. preflightバリデーション関数 🔵

**信頼性**: 🔵 *既存train_*.pyの前提条件より*

```python
# src/training/preflight.py
from pathlib import Path

class PreflightError(Exception):
    """Raised when training prerequisites are not met."""

def validate_training_prerequisites(cfg: BaselineConfig | TGLoRAConfig, config_path: str | Path) -> None:
    errors = []

    # データファイル存在確認
    for field_name in ["train_path", "valid_quick_path"]:
        path = Path(getattr(cfg.data, field_name))
        if not path.exists():
            errors.append(f"Data file not found: {path} (data.{field_name})")

    # run_dir書き込み確認
    run_dir = Path(cfg.logging.run_dir)
    run_dir.mkdir(parents=True, exist_ok=True)
    test_file = run_dir / ".write_test"
    try:
        test_file.write_text("ok")
        test_file.unlink()
    except OSError as e:
        errors.append(f"run_dir is not writable: {run_dir} ({e})")

    # max_seq_len確認（トークナイザー読み込み後に呼ぶ variant も定義）
    if cfg.data.max_seq_len < 32:
        errors.append(f"max_seq_len too small: {cfg.data.max_seq_len}")

    if errors:
        raise PreflightError("\n".join(errors))
```

**実装ファイル**: `src/training/preflight.py`（新規）

### 2. train_*.pyへの統合 🔵

**信頼性**: 🔵 *既存main()関数の構造より*

```python
# train_tg_lora.py main() 内
from src.training.config_schema import load_and_validate_config
from src.training.preflight import validate_training_prerequisites, PreflightError

def main() -> None:
    import argparse
    from omegaconf import OmegaConf
    from src.utils.logging import setup_logging

    parser = argparse.ArgumentParser(description="Run TG-LoRA training")
    parser.add_argument("--config", required=True, help="Path to YAML config")
    args = parser.parse_args()

    cfg = load_and_validate_config(args.config)  # スキーマ検証
    validate_training_prerequisites(cfg, args.config)  # ファイル存在等
    setup_logging()
    train_tg_lora(cfg)
```

**実装ファイル**: `src/training/train_tg_lora.py`, `src/training/train_baseline_qlora.py`
